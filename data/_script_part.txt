// ── 工具函数 ──────────────────────────────────────────────────
function openDB() {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open('KelHomeData', 1);
    req.onsuccess = () => resolve(req.result);
    req.onerror  = () => reject(req.error);
    req.onupgradeneeded = e => {
      const db = e.target.result;
      if (!db.objectStoreNames.contains('appdata'))
        db.createObjectStore('appdata');
    };
  });
}

function setStatus(msg, cls) {
  const el = document.getElementById('status');
  el.textContent = msg;
  el.className = cls;
}

// ── 导入 ──────────────────────────────────────────────────────
async function doImport() {
  setStatus('正在读取 KelHomeData…', 'info');
  try {
    const db   = await openDB();
    const tx   = db.transaction('appdata', 'readonly');
    const store = tx.objectStore('appdata');

    const existing = await new Promise((res, rej) => {
      const r = store.get('state');
      r.onsuccess = () => res(r.result);
      r.onerror   = () => rej(r.error);
    });

    if (!existing) {
      setStatus('❌ 读不到 Kel-Home 的数据（KelHomeData 里没有 state）。
请先在同一浏览器打开 Kel-Home，让它存一次数据，再回来导入。', 'err');
      return;
    }

    // 检查是否已导入
    if (existing.chats && existing.chats['imported-20260716']) {
      setStatus('⚠️ 这个对话已经导入过了（id: imported-20260716）。
不重复写入。如需覆盖请先在 Kel-Home 里删除那个对话。', 'info');
      return;
    }

    // 把新对话写进去
    const newState = JSON.parse(JSON.stringify(existing)); // 深拷贝
    if (!newState.chats) newState.chats = {};
    newState.chats['imported-20260716'] = {
      id: 'imported-20260716',
      title: '7月16日 局域网对话',
      messages: MSGS,
      lastTime: 1784214011958
    };
    newState._savedAt = Date.now();

    const wtx   = db.transaction('appdata', 'readwrite');
    const wstore = wtx.objectStore('appdata');
    await new Promise((res, rej) => {
      const r = wstore.put(newState, 'state');
      r.onsuccess = () => res();
      r.onerror   = () => rej(r.error);
    });

    const cnt = MSGS.length;
    setStatus(`✅ 导入成功！写入了 ${cnt} 条消息。
刷新 Kel-Home，在对话列表里找「7月16日 局域网对话」。`, 'ok');
  } catch (e) {
    setStatus('❌ 出错了：' + e.message, 'err');
  }
}
